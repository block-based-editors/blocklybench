Smooth Voxels Cheat Sheet 

texture

  id        = <textureid>                                  // The id/name of the texture
  cube      = {true|false}                                 // Whether this is a cube texture 
  image     = data:image/<type>;base64,<base64imagedata>   // Add images with the playground


light

  color     = <color> <float:strength>                     // All lights
  direction = <float:x> <float:y> <float:z>                // Directional lights
  position  = <float:x> <float:y> <float:z>                // Positional lights
  distance  = <float>                                      // How far the light works, Positional lights
  detail    = {0|1|2|3}                                    // The sphere detail, Positional lights


model

  size      = <int:x-size> <int:y-size> <int:z-size>       // The size of the voxel matrix
  origin    = -x x +x -y y +y -z z +z                      // Specify origin (one x, y and z component)
  scale     = <float:x> <float:y> <float:z>                // The scale of the voxels
  autoresize = {true|false}                                // Resize so the model fills the original boundaries
  rotation  = <float:x> <float:y <float:z>                 // Rotation around axes in degrees
  position  = <float:x> <float:y> <float:z>                // Moves the final model (if origin is not enough)
  flatten   = -x x +x -y y +y -z z +z                      // Flattens as if a part is cut off
  clamp     = -x x +x -y y +y -z z +z                      // Flattens with peripendicular sides
  skip      = -x x +x -y y +y -z z +z                      // Do not create these faces
  tile      = x y z                                        // Don't warp or scatter these model edges for tiling
  wireframe = {true|false}                                 // Make all materials wireframe
  ao        = <color> <float:maxDistance> <float:strength> // Ambient Occlusion
  aosides   = -x +x -y +y -z +z                            // Which sides are occluded by walls / floors, etc.
  shell     = <colorId> <float:distance> ...               // Shell for the model, colorId determines the material

material

  type      = {standard|phong|lambert|basic|toon|matcap|normal}   // The type of material
  lighting  = {flat|smooth|both}                           // Flat or smooth faces, or both when flattened/clamped
  roughness = <float>                                      // How rough the material looks
  metalness = <float>                                      // How metallic the material looks, (standard material)
  opacity   = <float>                                      // How opaque/transparent the material is
  transparent = {true|false}                               // Set when using transparent textures
  alphatest = <float>                                      // Solves many transparent texture issues (0.0 - 1.0)
  emissive  = <color> <float:intensity>                    // Emissive materials give light
  wireframe = {true|false}                                 // Makes this material wireframe
  fade      = {true|false}                                 // Whether voxel colors of this material will fade
  fog       = {true|false}                                 // Whether the material is affected by fog
  lights    = {true|false}                                 // Whether the material is affected by SVOX lights
  side      = {front|back|double}                          // Show the front, back or both sides of faces
  ao        = <color> <float:maxDistance> <float:strength> // Ambient Occlusion
  deform    = <int:count> <float:strength> <float:damping> // Deforms the voxels by means of vertex averaging
  warp      = <float:strength> <float:frequency>           // Warps the voxels, both parameters in voxels
  scatter   = <float:strength>                             // Scatters the vertices, in voxels
  flatten   = -x x +x -y y +y -z z +z | none               // Flattens as if a part is cut off
  clamp     = -x x +x -y y +y -z z +z | none               // Flattens with peripendicular sides
  skip      = -x x +x -y y +y -z z +z | none               // Do not create these faces
  map          = <textureid>                               // Use a texture (add images with the playground)
  normalmap    = <textureid>                               // Use a normal map
  roughnessmap = <textureid>                               // Use a roughness map
  metalnessmap = <textureid>                               // Use a metalness map
  emissivemap  = <textureid>                               // Use an emissive map
  maptransform = <width> <height> <xoffset> <yoffset> <rotation>   // Shift, scale or rotate the textures
  shell     = <colorId> <float:distance> ... | none        // Material shell, or 'none' for no shell
  colors    = A:#08F B:#FF8800 ...                         // Colors are #RGB or #RRGGBB
